on_actions = {
	on_startup = {
		#World state rules
		effect = {
			if = {
				limit = {
					has_game_rule = {
						rule = HGO_world_state
						option = BLANK_SLATE
					}
				}
				every_state = {
					set_demilitarized_zone = no
				}
				every_country = {
					limit = {
						NOT = {
							tag = WTR
						}
					}
					UNC = {
						annex_country = { target = PREV }
					}
				}
			}
		}
	}
	#Hyperborea color change
	on_ruling_party_change = {
		effect = {
			HYP = {
				drop_cosmetic_tag = yes
				if = {
					limit = {
						has_government = republican_fascism
					}
					set_cosmetic_tag = HYP_YAZOV
				}
				if = {
					limit = {
						has_government = democratic
					}
					set_cosmetic_tag = HYP_SHOTAKOVICH
				}
				if = {
					limit = {
						has_government = centrist_republic
					}
					set_cosmetic_tag = HYP_YELTSIN
				}
				if = {
					limit = {
						has_government = neutrality
					}
					set_cosmetic_tag = HYP_ZHUKOV
				}
				if = {
					limit = {
						has_government = liberal_republic
					}
					set_cosmetic_tag = HYP_LIKHACHEV
				}
				if = {
					limit = {
						has_government = fascism
					}
					set_cosmetic_tag = HYP_RODZAEVSKY
				}
				if = {
					limit = {
						has_government = monarchism
					}
					set_cosmetic_tag = HYP_TABORITSKY
				}
				if = {
					limit = {
						has_government = left_anarchism
					}
					set_cosmetic_tag = HYP_ANARCHY
				}
				if = {
					limit = {
						has_government = right_anarchism
					}
					set_cosmetic_tag = HYP_GUMILNOV
				}
				if = {
					limit = {
						has_government = vanguardism
					}
					set_cosmetic_tag = HYP_ZHDANOV
				}
			}
		}
	}
}
